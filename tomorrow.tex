\newcommand{\yesterday}{%
	\DTMsavenow{mydate}
	\DTMsaveddateoffsettojulianday{mydate}{-1}{\myct}
	\DTMsavejulianday{mydate}{\number\myct}
	\DTMusedate{mydate}
}
\newcommand{\myday}[1]{%
	\DTMsavenow{mydate}
	\DTMsaveddateoffsettojulianday{mydate}{#1}{\myct}
	\DTMsavejulianday{mydate}{\number\myct}
	\DTMusedate{mydate}
}
\newcommand{\numofday}{%
	\DTMsavenow{mydate}
	\DTMsaveddateoffsettojulianday{mydate}{1}{\myct}
	\DTMsavejulianday{mydate}{\number\myct}
	\DTMsavedate{mydate2}{\DTMfetchyear{mydate}-01-01}
	\DTMsaveddatediff{mydate}{mydate2}{\myct}
		\number\myct
}
